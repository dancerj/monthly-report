\section{Debian と Debian JPについて}

\begin{frame}
  \begin{center}\Huge{Debian とは？}\end{center}
\end{frame}


\begin{frame}{Debian とは？}
  
\begin{itemize}
\item \url{https://www.debian.org/}
\item Debianはフリー/オープンなオペレーティングシステム (OS)を作成しようとするボランティアベースのプロジェクト。
\item 自分たちの考えるフリーという言葉に関する定義、開発目的、パッケージングポリシーを厳格に決めている。
\item 世界中に1000人以上の開発者がおり、他のディストリビューションのベースとして採用されている。
\item 約2年毎にリリースが行われ、多くのパッケージとアーキテクチャをサポートしている。
\item 年に一度、Debconfという開発者が集うイベントがある。Debconf18は7/21-7/27に台湾で開催予定。
\item 上記のような特徴から様々なところで利用されているLinuxディストリビューションである。
\end{itemize}

\end{frame}

\section{Debianとwebサーバ}

\begin{frame}
  \begin{center}\Huge{Debianとwebサーバ}\end{center}
\end{frame}

\begin{frame}{Debianのwebサーバで言われること}% [containsverbatim]
  
\begin{itemize}
\item webサーバ用途でDebianはあまり使われていないのではないか
\item webサーバがいろいろあってどのパッケージをインストールすればいいかわからない
\item apacheやnginxのディレクトリ構造がRHELやtarballでインストールしたときと異なるためよくわからない
\end{itemize}

そんな疑問にお答えするため、今回はdebian 9 の apache2 及び pythonのWebアプリケーションサーバ"WSGI"の設定についてお話しします。

\end{frame}


\begin{frame}{webサーバになぜDebianを選択するのか}% [containsverbatim]
\begin{itemize}
  \item Linuxのwebサーバの稼働OSシェア\footnote{\url{https://w3techs.com/technologies/details/os-linux/all/all}}
  \begin{itemize}
    \item 1位 Ubuntu 39.1 \%
    \item 2位 Debian 30.8 \%
    \item 3位 CentOS 20.6 \%
    \item 4位 Red Hat 2.9 \%
  \end{itemize}

  \item 月100万アクセス以上のwebサイトのwebサーバシェア\footnote{\url{https://news.netcraft.com/archives/2018/01/19/january-2018-web-server-survey.html}}
  \begin{itemize}
    \item 1位 Apache 37.03 \% %% 2012年くらいから下がり始めた
    \item 2位 nginx 30.46 \% %% apacheの減少がnginxに移っている
    \item 3位 Microsoft IIS Web Server 9.57 \%      
  \end{itemize}
\end{itemize}

\end{frame}


\begin{frame}{Debianで提供しているwebサーバ}% [containsverbatim]

\begin{itemize}
  \item apache2 (apache-2.4.25)
  \item nginx (nginx-1.10.3、stretch-backportsはnginx-1.13.3)
  \item lighttpd (lighttpd-1.4.45)
\end{itemize}

\end{frame}


\begin{frame}{Debainで提供しているWSGIサーバ}% [containsverbatim]
  
\begin{itemize}
\item WSGI = Web Server Gateway Interface
\item WSGIを実装したwebアプリケーションサーバ
  \begin{itemize}
  \item mod\_wsgi (libapache2-mod-wsgi、libapache2-mod-wsgi-py3パッケージ)
    \begin{itemize}
    \item apache2の拡張モジュール。apache2のみ利用可能。
    \end{itemize}
  \item uwsgi (uwsgiパッケージ)、gunicorn
    \begin{itemize}
    \item wsgiプロトコルで待ち受けするデーモン。nginx、apache2で利用可能。
    \end{itemize}
  \item gunicorn (python-gunicorn、python3-gunicornパッケージ)
    \begin{itemize}
    \item wsgiプロトコルで待ち受けするデーモン。nginx、apache2で利用可能。
    \end{itemize}
  \end{itemize}
\item 性能ベンチマークの例（2016年5月）\footnote{\url{https://blog.appdynamics.com/engineering/a-performance-analysis-of-python-wsgi-servers-part-2/}}
\end{itemize}

\end{frame}

\begin{frame}[containsverbatim]{mod\_wsgiの設定（1）}

\end{commandline}

\end{frame}


\begin{frame}[containsverbatim]{mod\_wsgiの設定（2）}

mod\_wsgiにおけるアプリケーションの動作モードは2種類あります\footnote{\url{http://modwsgi.readthedocs.io/en/latest/user-guides/reloading-source-code.html}}
  
\begin{itemize}
  \item Embedded Mode
  \item Daemon Mode
\end{itemize}

\begin{table}[htb]
  \caption{mod\_wsgiの動作モードの比較}
  \begin{tabular}{|l|c|c|}
    \hline
    ― & Embedded Mode & Daemon Mode \\
    \hline \hline
    メモリ使用量 & 大 & 中 \\
    \hline
    CPU使用量 & 大 & 中 \\
    \hline
    性能 & 高 & 中 \\
    \hline
    アプリ再読み込み & restart apache2 & \shortstack{reload apache2 \\ or touch wsgi.py} \\
    \hline
    apache MPM & prefork & \shortstack{prefork \\ worker \\ event} \\
    \hline
  \end{tabular}
\end{table}

\end{frame}


\begin{frame}[containsverbatim]{mod\_wsgiの設定（3）}

アプリケーション固有の設定（daemon modeの設定）

\begin{commandlinesmall}
# vi /etc/apache2/conf-available/wsgi_mysite.conf

WSGIScriptAlias /mysite /var/www/django-tutorial/mysite/mysite/wsgi.py

WSGIDaemonProcess mysite user=www-data group=www-data \
processes=2 threads=64 maximum-requests=256 \
python-path=/var/www/django-tutorial/mysite \
display-name=%{GROUP}

WSGIProcessGroup  mysite
\end{commandlinesmall}

% 設定として紹介するか検討
%
% Alias /static/polls ``/var/www/django-tutorial/mysite/polls/static/''
% <Directory ``/var/www/django-tutorial/mysite/polls/static/''>
% Require all granted
% </Directory>

\begin{itemize}
\item virtualenvを使う場合、WSGIDaemonProcess に "python-home" を指定する
\item WSGIDaemonProcessのパラメータはURLを参照\footnote{\url{http://modwsgi.readthedocs.io/en/latest/configuration-directives/WSGIDaemonProcess.html}}
\end{itemize}

\end{frame}


\begin{frame}[containsverbatim]{apache2の設定}

Debianの/etc/apache2にディレクトリ構成

\begin{itemize}
  \item apache2.conf, envvars, magic, ports.conf
    \begin{itemize}
    \item 設定や環境変数を定義するファイル
    \end{itemize}
  \item conf-available, conf-enabled ディレクトリ
    \begin{itemize}
    \item 設定ファイル と 有効/無効の切り替え設定 を保持
    \item a2enconf と a2disconf で有効/無効を切り替え
    \end{itemize}
  \item mods-available, mods-enabled ディレクトリ
    \begin{itemize}
    \item 拡張モジュールの設定ファイル と 有効/無効の切り替え設定 を保持
    \item a2enmod と a2dismod で有効/無効を切り替え
    \end{itemize}
  \item sites-available, sites-enabled ディレクトリ
    \begin{itemize}
    \item VirtualHostの設定ファイル と 有効/無効の切り替え設定 を保持
    \item a2ensite と a2dissite で有効/無効を切り替え
    \end{itemize}
\end{itemize}
  
\end{frame}
